91  ARFaceTrackingConfiguration * config = [ARFaceTrackingConfiguration new];
92 
93  if(state.useLightEstimation){
94  config.lightEstimationEnabled = YES;
95  }
96 
97  config.worldAlignment = state.worldAlignment;
98 
99  if(state.delegateClass != NULL){
100  session.delegate = state.delegateClass;
101 
102  // note that audio is only available when you use a delegate class.
103  if(state.useAudio){
104  config.providesAudioData = YES;
105  }
106  }
107 
108  [session runWithConfiguration:config];
109 
110  return session;
111  }else{
112  NSLog(@"Unable to use face tracking configuration, defaulting to a more standard config.");
113  }
114  }
115 
116 
117  // if face tracking is not available, should pass through to here where we
118  // figure out regular configuration, starting with determining if we can do plane detection.
119  if([ARWorldTrackingConfiguration isSupported]){
120 
121  ARWorldTrackingConfiguration * config = [ARWorldTrackingConfiguration new];
122 
123  if(state.usePlaneTracking){
124  // ofLog()<<"Using plane tracking";
125  config.planeDetection = state.planeDetectionType;
126  }
127 
128  if(state.useLightEstimation){
129  config.lightEstimationEnabled = YES;
130  }
131 
132  config.worldAlignment = state.worldAlignment;
133 
134  if(state.delegateClass != NULL){
135  session.delegate = state.delegateClass;
136 
137  // note that audio is only available when you use a delegate class.
138  if(state.useAudio){
139  config.providesAudioData = YES;
140  }
141  }
142 
143  [session runWithConfiguration:config];
144

146  }else {
